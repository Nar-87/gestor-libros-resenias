Create table if not exists tbresenia  (
	idresenia int primary key AUTO_INCREMENT,
    titulores varchar(50),
	puntosres int not null,
	contenidores text,
	fechares date not null,
	idlibro int not null, 
	idusu int not null,
	foreign key (idlibro) references tblibro(idlibro),
	foreign key (idusu) references tbusuario(idusu)
);

Create table if not exists tbtaglibro (
	idlibro int not null,
	idtag int not null,
	primary key(idlibro, idtag),
	foreign key (idlibro) references tblibro(idlibro),
	foreign key (idtag) references tbtag(idtag)
);
